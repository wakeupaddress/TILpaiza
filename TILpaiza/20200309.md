```
<?php
    // 標準入力から値を取得してinput_lineに入れる
    
    $input_line=fgets(STDIN);
    
    $number=fgets(STDIN);

    $before=explode("",$input_line);
    
    if($number=1;){
            str_replace($before[0],"",$before);
            echo $before;
    }
    elseif($number=2;){
            str_replace("$before[1]","",$before);
            echo $before;
    }
    elseif($number=3;){
            str_replace("$before[2]","",$before);
            echo $before;
    }
    elseif($number=4;){
            str_replace("$before[3]","",$before);
            echo $before;
    }
    elseif($number=5;){
            str_replace("$before[4]","",$before);
            echo $before;
    }
    
?>
```
