D005:等差数列

```
<?php
    // 自分の得意な言語で
    // Let's チャレンジ！！

    $input_line = fgets(STDIN);
    
    $replace = str_replace(PHP_EOL,"",$input_line);
    
    $explode = explode(" ",$replace);
    
    for($i=0; $i<10; $i++){

      if($i<=8){
      echo $explode[0]+($i*$explode[1])." ";
      }
      else{
      echo $explode[0]+($i*$explode[1]);
      }
      
    }
?>
```
